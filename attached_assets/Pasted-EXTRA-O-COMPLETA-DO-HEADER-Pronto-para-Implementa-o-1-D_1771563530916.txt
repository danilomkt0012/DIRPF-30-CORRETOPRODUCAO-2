// === COMPONENTE DO HEADER ===
function RFBHeader({ displayFirstName = "Entrar" }: { displayFirstName?: string }) {
  const [showFuncMenu, setShowFuncMenu] = useState(false);
  const [showSearch, setShowSearch] = useState(false);
  const [showSideMenu, setShowSideMenu] = useState(false);
  const [expandedMenus, setExpandedMenus] = useState<Record<string, boolean>>({});
  const [menuFilter, setMenuFilter] = useState("");
  const funcRef = useRef<HTMLDivElement>(null);
  const funcPanelRef = useRef<HTMLDivElement>(null);
  const headerRef = useRef<HTMLElement>(null);
  const sideMenuRef = useRef<HTMLDivElement>(null);
  const toggleSubmenu = useCallback((key: string) => {
    setExpandedMenus(prev => ({ ...prev, [key]: !prev[key] }));
  }, []);
  // Fechar menus ao clicar fora ou pressionar Escape
  useEffect(() => {
    const handleClickOutside = (e: MouseEvent) => {
      if (
        funcRef.current && !funcRef.current.contains(e.target as Node) &&
        (!funcPanelRef.current || !funcPanelRef.current.contains(e.target as Node))
      ) {
        setShowFuncMenu(false);
      }
      if (sideMenuRef.current && !sideMenuRef.current.contains(e.target as Node)) {
        setShowSideMenu(false);
      }
    };
    const handleKeyDown = (e: KeyboardEvent) => {
      if (e.key === 'Escape') {
        setShowFuncMenu(false);
        setShowSearch(false);
        setShowSideMenu(false);
      }
    };
    document.addEventListener('mousedown', handleClickOutside);
    document.addEventListener('keydown', handleKeyDown);
    return () => {
      document.removeEventListener('mousedown', handleClickOutside);
      document.removeEventListener('keydown', handleKeyDown);
    };
  }, []);
  // Filtro do menu lateral
  const filterMenu = (items: any[], filter: string): any[] => {
    if (!filter) return items;
    const lf = filter.toLowerCase();
    return items.reduce((acc: any[], item: any) => {
      if (item.label.toLowerCase().includes(lf)) {
        acc.push(item);
      } else if (item.children) {
        const filtered = filterMenu(item.children, filter);
        if (filtered.length > 0) acc.push({ ...item, children: filtered });
      }
      return acc;
    }, []);
  };
  // Renderizador recursivo dos itens do menu
  const renderMenuItem = (item: any, key: string, level: number = 0) => {
    const hasChildren = item.children && item.children.length > 0;
    const isExpanded = expandedMenus[key];
    const lf = menuFilter.toLowerCase();
    const matchesFilter = lf && item.label.toLowerCase().includes(lf);
    return (
      <div key={key} className="side-menu-node">
        <button
          className={`side-menu-item level-${level}${matchesFilter ? ' highlight' : ''}`}
          onClick={() => hasChildren ? toggleSubmenu(key) : undefined}
          aria-expanded={hasChildren ? isExpanded : undefined}
        >
          {item.icon && level === 0 && <span className="material-icons-round side-menu-icon">{item.icon}</span>}
          <span className="side-menu-label">{item.label}</span>
          {hasChildren && (
            <span className="material-icons-round side-menu-arrow" style={isExpanded ? { transform: 'rotate(180deg)' } : undefined}>expand_more</span>
          )}
        </button>
        {hasChildren && isExpanded && (
          <div className="side-menu-children">
            {item.children.map((child: any, i: number) => renderMenuItem(child, `${key}-${i}`, level + 1))}
          </div>
        )}
      </div>
    );
  };
  return (
    <>
      {/* ===== HEADER PRINCIPAL ===== */}
      <header className="rfb-header" ref={headerRef}>
        <div className="rfb-header-container">
          {/* Linha superior: Logo + Ações */}
          <div className="rfb-header-top">
            <div className="rfb-header-logo">
              <a href="https://servicos.receitafederal.gov.br" rel="noopener noreferrer" target="_self">
                <img src={RFB_LOGO} alt="Receita Federal" />
              </a>
            </div>
            <div className="header-actions">
              {/* Botão Funcionalidades */}
              <div className="header-functions" ref={funcRef}>
                <button
                  type="button"
                  aria-label="Abrir Funcionalidades do Sistema"
                  title="Abrir Funcionalidades do Sistema"
                  aria-expanded={showFuncMenu}
                  className={`header-icon-btn${showFuncMenu ? ' active' : ''}`}
                  onClick={() => { setShowFuncMenu(v => !v); setShowSearch(false); }}
                >
                  <i aria-hidden="true" className="fas fa-th" />
                </button>
              </div>
              {/* Divisor vertical */}
              <div className="header-divider" />
              {/* Botão Busca */}
              <div className="header-search-trigger">
                <button
                  type="button"
                  aria-label="Abrir Busca"
                  title="Abrir Busca"
                  className={`header-icon-btn${showSearch ? ' active' : ''}`}
                  onClick={() => { setShowSearch(v => !v); setShowFuncMenu(false); }}
                >
                  <i aria-hidden="true" className="fas fa-search" />
                </button>
              </div>
              {/* Botão Login/Usuário */}
              <div className="header-login">
                <button type="button" className="br-sign-in" aria-label="Entrar">
                  <div className="br-sign-in-content">
                    <i className="fas fa-user" aria-hidden="true" />
                    <span>{displayFirstName}</span>
                  </div>
                </button>
              </div>
            </div>
          </div>
          {/* Painel de Funcionalidades (Ajuda, Cookies, Contraste) */}
          {showFuncMenu && (
            <div className="func-panel" ref={funcPanelRef}>
              <div className="func-panel-header">
                <span className="func-panel-title">Funcionalidades do Sistema</span>
              </div>
              <div className="func-panel-item has-chevron">
                <button type="button" title="Ajuda" className="func-panel-btn">
                  <span className="func-panel-icon circle-border"><i aria-hidden="true" className="fas fa-question" /></span>
                  <span className="func-panel-label">Ajuda</span>
                  <i aria-hidden="true" className="fas fa-angle-right func-panel-chevron" />
                </button>
              </div>
              <div className="func-panel-item">
                <button type="button" title="Cookies" className="func-panel-btn">
                  <span className="func-panel-icon"><i aria-hidden="true" className="fas fa-cookie-bite" /></span>
                  <span className="func-panel-label">Cookies</span>
                </button>
              </div>
              <div className="func-panel-item">
                <button type="button" title="Alternar alto contraste" className="func-panel-btn">
                  <span className="func-panel-icon"><i aria-hidden="true" className="fas fa-adjust" /></span>
                  <span className="func-panel-label">Contraste</span>
                </button>
              </div>
            </div>
          )}
          {/* Linha inferior: Menu hamburger + Título */}
          {!showFuncMenu && (
            <div className="rfb-header-bottom">
              <div className="header-menu">
                <div className="header-menu-trigger">
                  <button
                    type="button"
                    aria-label="Menu"
                    className="header-icon-btn"
                    onClick={() => { setShowSideMenu(v => !v); setShowFuncMenu(false); setShowSearch(false); }}
                  >
                    <i className="fa fa-bars" />
                  </button>
                </div>
                <div className="header-info" style={{ paddingRight: '92px' }}>
                  <div className="header-title"><a>Serviços do Contribuinte</a></div>
                </div>
              </div>
            </div>
          )}
          {/* Barra de busca expansível */}
          {showSearch && (
            <div className="header-search-row">
              <div className="header-search-input-wrap">
                <input type="text" placeholder="O que você procura?" autoFocus />
                <button type="button" className="header-search-btn" aria-label="Pesquisar" title="Pesquisar">
                  <i className="fas fa-search" />
                </button>
                <button type="button" className="header-search-close" onClick={() => setShowSearch(false)} aria-label="Fechar busca">
                  <i className="fas fa-times" />
                </button>
              </div>
            </div>
          )}
        </div>
      </header>
      {/* ===== MENU LATERAL (Side Menu) ===== */}
      {showSideMenu && <div className="side-menu-overlay" onClick={() => setShowSideMenu(false)} />}
      <div className={`side-menu${showSideMenu ? ' open' : ''}`} ref={sideMenuRef}>
        <div className="side-menu-header">
          <img src={RFB_LOGO} alt="Receita Federal" className="side-menu-logo" />
          <button className="side-menu-close" onClick={() => setShowSideMenu(false)} aria-label="Fechar menu">
            <span className="material-icons-round">close</span>
          </button>
        </div>
        <div className="side-menu-filter-wrap">
          <input
            type="text"
            placeholder="Filtrar"
            className="side-menu-filter"
            value={menuFilter}
            onChange={(e) => setMenuFilter(e.target.value)}
          />
          <span className="material-icons-round side-menu-filter-icon" onClick={() => setMenuFilter("")}>backspace</span>
        </div>
        <div className="side-menu-content">
          {filterMenu(MENU_DATA, menuFilter).map((item, i) => renderMenuItem(item, `m${i}`, 0))}
        </div>
        <div className="side-menu-footer">
          <div className="menu-links">
            <a href="https://www.gov.br/receitafederal/pt-br/centrais-de-conteudo/download" target="_blank" rel="noopener noreferrer" className="side-menu-footer-link">
              <span>Download de Programas</span>
              <span className="material-icons-round external-icon">open_in_new</span>
            </a>
          </div>
          <div className="social-network">
            <span className="social-network-title">Redes Sociais</span>
            <div className="social-icons">
              <a href="https://www.facebook.com/receitafederaloficial" target="_blank" rel="noopener noreferrer" aria-label="Facebook">
                <svg viewBox="0 0 24 24" width="16" height="16" fill="currentColor"><path d="M22 12c0-5.523-4.477-10-10-10S2 6.477 2 12c0 4.991 3.657 9.128 8.438 9.878v-6.987h-2.54V12h2.54V9.797c0-2.506 1.492-3.89 3.777-3.89 1.094 0 2.238.195 2.238.195v2.46h-1.26c-1.243 0-1.63.771-1.63 1.562V12h2.773l-.443 2.89h-2.33v6.988C18.343 21.128 22 16.991 22 12z"/></svg>
              </a>
              <a href="https://www.instagram.com/receita_federal" target="_blank" rel="noopener noreferrer" aria-label="Instagram">
                <svg viewBox="0 0 24 24" width="16" height="16" fill="currentColor"><path d="M12 2.163c3.204 0 3.584.012 4.85.07 3.252.148 4.771 1.691 4.919 4.919.058 1.265.069 1.645.069 4.849 0 3.205-.012 3.584-.069 4.849-.149 3.225-1.664 4.771-4.919 4.919-1.266.058-1.644.07-4.85.07-3.204 0-3.584-.012-4.849-.07-3.26-.149-4.771-1.699-4.919-4.92-.058-1.265-.07-1.644-.07-4.849 0-3.204.013-3.583.07-4.849.149-3.227 1.664-4.771 4.919-4.919 1.266-.057 1.645-.069 4.849-.069zM12 0C8.741 0 8.333.014 7.053.072 2.695.272.273 2.69.073 7.052.014 8.333 0 8.741 0 12c0 3.259.014 3.668.072 4.948.2 4.358 2.618 6.78 6.98 6.98C8.333 23.986 8.741 24 12 24c3.259 0 3.668-.014 4.948-.072 4.354-.2 6.782-2.618 6.979-6.98.059-1.28.073-1.689.073-4.948 0-3.259-.014-3.667-.072-4.947-.196-4.354-2.617-6.78-6.979-6.98C15.668.014 15.259 0 12 0zm0 5.838a6.162 6.162 0 100 12.324 6.162 6.162 0 000-12.324zM12 16a4 4 0 110-8 4 4 0 010 8zm6.406-11.845a1.44 1.44 0 100 2.881 1.44 1.44 0 000-2.881z"/></svg>
